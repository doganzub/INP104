class Personel():

    def __init__(self, isim):
        self.isim = isim

        self.personel.append(self.isim)

class Mesai(Personel):
    maas = 3000
    eksaat = 50
    toplam = []

    def __init__(self, isim, saat, tutar):
        super().__init__(isim)
        self.saat = saat
        self.tutar = tutar
        self.hesap()
        self.tutar_ekle()

    def hesap(self, maas=None):
        if self.saat <= 40:
            self.maas = maas
            return maas

        if self.saat > 40:
            self.tutar = self.maas + (self.eksaat * (self.saat - 40))
            return self.tutar

    def tutar_ekle(self):
        self.toplam.append(self.tutar)
        print('{} adlı personele alacak eklendi'.format(self.isim))

    @classmethod # c1 fonksiyonu sınıf metoduna dönüştürme işlevi görüyor
    def c1(cls, maas=None):
        cls.maas = maas
        return cls.maas

    @classmethod
    def c2(cls, eksaat=None):
        cls.eksaat = eksaat
        return cls.eksaat

        self.isim = isim
        self.statu = statu
        self.yas = yas
        self.personel_ekle()

    def personel_ekle(self):
        self.personel.append(self._no)
        self.personel.append(self.isim)
        self.personel.append(self.statu)
        self.personel.append(self.yas)


    #@property ile salt okunur: ('self.no' AttributeError: can't set attribute)
    #@property ile değişiklik yapılabilir 1.yol (__init__ altında self._no)
    @property
    def no(self):
        return self._no

    #@property ile değişiklik yapılabilir 2.yol (__init__ altındaki 'self.no')
    @no.setter
    def no(self,yeni_deger):
        self._no = yeni_deger
        return self._no
